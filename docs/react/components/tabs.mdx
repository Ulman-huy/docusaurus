# Tabs

### Example

```jsx
import Tabs from "./Tabs";

const TabsExample = () => {
  const items = [
    {
      key: "1",
      label: "Tab 1",
      content: "Content 1",
    },
    {
      key: "2",
      label: "Tab 2",
      content: "Content 2",
    },
  ];

  return <Tabs items={items} defaultActiveKey="1" />;
};
export default TabsExample;
```

### Code

```jsx title="Tabs/index.tsx"
import clsx from "clsx";
import { ReactNode, useEffect, useState } from "react";

interface TabsProps {
  defaultActiveKey?: string;
  idChange?: any;
  items: {
    key: string;
    label: string;
    content: ReactNode;
  }[];
}

const Tabs: React.FC<TabsProps> = ({
  defaultActiveKey,
  idChange = "",
  items = [],
}) => {
  const [active, setActive] = useState(defaultActiveKey ?? items[0].key);
  const [index, setIndex] = useState(() => {
    let result = 0;
    items.map((item, index) => {
      if (item.key == defaultActiveKey) {
        result = index;
      }
    });
    return result;
  });

  useEffect(() => {
    if (defaultActiveKey) {
      setActive(defaultActiveKey);
      items.map((item, index) => {
        if (item.key == defaultActiveKey) {
          setIndex(index);
        }
      });
    }
  }, [defaultActiveKey, idChange]);

  return (
    <>
      <div className="flex justify-center">
        {items.map((item, index) => (
          <div
            key={item.key}
            onClick={() => {
              setActive(item.key);
              setIndex(index);
            }}
            className={clsx(
              "w-full flex justify-center cursor-pointer p-2.5 pb-2 text-sm transition",
              active == item.key
                ? "text-green-50 font-bold"
                : "text-white/60 font-medium"
            )}
          >
            <span>{item.label}</span>
          </div>
        ))}
      </div>
      <div style={{ height: 2, width: "100%", background: "#FFFFFF1A" }}>
        <div
          // style={{
          //  width: `${100 / items.length}%`,
          //  transform: `translateX(${100 * index}%)`,
          //  height: "100%",
          //  background: "#009874",
          //  transitionDuration: "200ms",
          //  transition: "transform cubic-bezier(0.4, 0, 0.2, 1) 150ms",
          // }}
        ></div>
      </div>
      {items[index].content}
    </>
  );
};

export default Tabs;

```
